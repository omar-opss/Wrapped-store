${item.name}

${item.price} EGP

`; cartItemsDiv.appendChild(itemDiv); }); totalSpan.textContent = total.toFixed(2) + " EGP"; } function removeItem(index) { cart.splice(index, 1); localStorage.setItem("cart", JSON.stringify(cart)); renderCart(); } function checkout() { window.location.href = "payment.html"; } renderCart();
